-- Função para criar o botão de minimizar
local function createMinimizeButton(frame)
    local minimizeButton = Instance.new("TextButton")
    minimizeButton.Size = UDim2.new(0, 50, 0, 50)
    minimizeButton.Position = UDim2.new(1, -55, 0, 5)
    minimizeButton.Text = "9"
    minimizeButton.TextColor3 = Color3.fromRGB(255, 0, 0)
    minimizeButton.BackgroundTransparency = 1
    minimizeButton.Parent = frame

    local minimized = false
    minimizeButton.MouseButton1Click:Connect(function()
        minimized = not minimized
        if minimized then
            frame.Size = UDim2.new(0, 300, 0, 50)
        else
            frame.Size = UDim2.new(0, 300, 0, 350)
        end
    end)
end

-- Função para obter inventário do jogador
local function getInventory(playerName)
    local player = game.Players:FindFirstChild(playerName)
    if player then
        local inventoryItems = {}
        for _, item in ipairs(player.Backpack:GetChildren()) do
            table.insert(inventoryItems, item.Name)
        end
        if #inventoryItems == 0 then
            return "Este jogador não tem nada no inventário"
        else
            return table.concat(inventoryItems, ", ")
        end
    else
        return "Jogador não encontrado"
    end
end

-- Conectar função ao campo de texto
nameBox.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        local playerName = nameBox.Text
        inventoryList.Text = getInventory(playerName)
    end
end)
